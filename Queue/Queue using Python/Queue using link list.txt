// Queue using link list
class node:
        def __init__(self):
            self.data=int(input('Enter value'))
            self.next=None
class Queue:
    def insert(self):
        global start,current
        temp=node()
        if(start==None):
            start=temp
            current=temp
        else:
            current.next=temp
            current=temp
            
    def delete(self):
         global start,current
         temp=start
         start=start.next
         print(temp.data,'is deleted')
         del(temp)
    def display(self):
         temp=start
         while(temp!=None):
              print(temp.data)
              temp=temp.next
         


start=None
current=None
while True:
    obj=Queue()
    ch =int(input('1 insert ||2 delete || 3 Display'))
    if ch==1:
        obj.insert()
    elif ch==2:
        obj.delete()
    elif ch==3:
        
        obj.display()
    else:
        break
